onNewTurn gets called when all players have played 1 round.

@override
onNewTurn() {
  // This is an example project:
  // every turn all the tile prices raise by 5%.
  Game.data.gmap.forEach((Tile tile) {
    raisePrice(tile);
  });
  // run Game.save() if you have changed data.
  Game.save();
}
